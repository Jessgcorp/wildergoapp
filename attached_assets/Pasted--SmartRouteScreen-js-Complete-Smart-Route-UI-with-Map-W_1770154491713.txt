// SmartRouteScreen.js
// Complete Smart Route UI with Map, Weather, and Road Risk

import React, { useState, useEffect } from 'react';
import {
  View,
  Text,
  StyleSheet,
  TouchableOpacity,
  ScrollView,
  ActivityIndicator,
  Dimensions
} from 'react-native';
import MapView, { Polyline, Marker, PROVIDER_GOOGLE } from 'react-native-maps';
import { getSmartRoute } from './smartRouteService';

const { width, height } = Dimensions.get('window');

export default function SmartRouteScreen() {
  const [smartRoute, setSmartRoute] = useState(null);
  const [loading, setLoading] = useState(false);
  const [selectedVehicle, setSelectedVehicle] = useState('van');
  const [mapRegion, setMapRegion] = useState({
    latitude: 39.7392,
    longitude: -104.9903,
    latitudeDelta: 0.8,
    longitudeDelta: 0.8,
  });

  // Sample route: Denver to Boulder
  const origin = { latitude: 39.7392, longitude: -104.9903 };
  const destination = { latitude: 40.0150, longitude: -105.2705 };

  useEffect(() => {
    loadSmartRoute();
  }, []);

  const loadSmartRoute = async () => {
    setLoading(true);
    try {
      const route = await getSmartRoute(origin, destination);
      if (route.success) {
        setSmartRoute(route);
        
        // Center map on route
        const midLat = (origin.latitude + destination.latitude) / 2;
        const midLon = (origin.longitude + destination.longitude) / 2;
        setMapRegion({
          latitude: midLat,
          longitude: midLon,
          latitudeDelta: 0.8,
          longitudeDelta: 0.8,
        });
      }
    } catch (error) {
      console.error('Error loading Smart Route:', error);
    } finally {
      setLoading(false);
    }
  };

  const getRoadRiskColor = (level) => {
    switch (level) {
      case 'safe': return '#2D5A3D'; // Forest Green
      case 'caution': return '#F2A154'; // Orange
      case 'warning': return '#D94848'; // Red
      default: return '#2D5A3D';
    }
  };

  const getWeatherEmoji = (condition) => {
    const emojiMap = {
      'Clear': '☀️',
      'Clouds': '☁️',
      'Rain': '🌧️',
      'Snow': '❄️',
      'Thunderstorm': '⛈️',
      'Drizzle': '🌦️',
      'Mist': '🌫️',
      'Fog': '🌫️',
    };
    return emojiMap[condition] || '🌤️';
  };

  if (loading) {
    return (
      <View style={styles.loadingContainer}>
        <ActivityIndicator size="large" color="#C65D3B" />
        <Text style={styles.loadingText}>Calculating Smart Route...</Text>
      </View>
    );
  }

  return (
    <View style={styles.container}>
      {/* Header */}
      <View style={styles.header}>
        <Text style={styles.headerTitle}>Smart Route</Text>
        <Text style={styles.headerSubtitle}>AI-Powered Navigation</Text>
      </View>

      {/* Map */}
      <MapView
        provider={PROVIDER_GOOGLE}
        style={styles.map}
        region={mapRegion}
        onRegionChangeComplete={setMapRegion}
      >
        {/* Route Polyline */}
        {smartRoute && smartRoute.coordinates && (
          <Polyline
            coordinates={smartRoute.coordinates}
            strokeColor="#C65D3B" // Burnt Sienna (brand color!)
            strokeWidth={5}
            lineDashPattern={[0]}
          />
        )}

        {/* Start Marker with Weather */}
        {smartRoute && (
          <Marker
            coordinate={origin}
            title="Start: Denver"
            description={`${smartRoute.weather.start.temperature}°F - ${smartRoute.weather.start.description}`}
          >
            <View style={styles.markerContainer}>
              <View style={[styles.marker, styles.startMarker]}>
                <Text style={styles.markerEmoji}>
                  {getWeatherEmoji(smartRoute.weather.start.condition)}
                </Text>
                <Text style={styles.markerTemp}>
                  {smartRoute.weather.start.temperature}°
                </Text>
              </View>
              <View style={styles.markerArrow} />
            </View>
          </Marker>
        )}

        {/* Middle Marker with Weather */}
        {smartRoute && smartRoute.coordinates && (
          <Marker
            coordinate={smartRoute.coordinates[Math.floor(smartRoute.coordinates.length / 2)]}
            title="Midpoint"
            description={`${smartRoute.weather.middle.temperature}°F - ${smartRoute.weather.middle.description}`}
          >
            <View style={styles.markerContainer}>
              <View style={[styles.marker, styles.middleMarker]}>
                <Text style={styles.markerEmoji}>
                  {getWeatherEmoji(smartRoute.weather.middle.condition)}
                </Text>
                <Text style={styles.markerTemp}>
                  {smartRoute.weather.middle.temperature}°
                </Text>
              </View>
              <View style={styles.markerArrow} />
            </View>
          </Marker>
        )}

        {/* End Marker with Weather */}
        {smartRoute && (
          <Marker
            coordinate={destination}
            title="Destination: Boulder"
            description={`${smartRoute.weather.end.temperature}°F - ${smartRoute.weather.end.description}`}
          >
            <View style={styles.markerContainer}>
              <View style={[styles.marker, styles.endMarker]}>
                <Text style={styles.markerEmoji}>
                  {getWeatherEmoji(smartRoute.weather.end.condition)}
                </Text>
                <Text style={styles.markerTemp}>
                  {smartRoute.weather.end.temperature}°
                </Text>
              </View>
              <View style={styles.markerArrow} />
            </View>
          </Marker>
        )}
      </MapView>

      {/* Bottom Info Card */}
      {smartRoute && (
        <View style={styles.infoCard}>
          {/* Route Info */}
          <View style={styles.routeInfo}>
            <Text style={styles.distance}>{smartRoute.distance}</Text>
            <Text style={styles.duration}>{smartRoute.duration}</Text>
          </View>

          {/* Road Risk Indicator */}
          <View style={[
            styles.roadRiskBadge,
            { backgroundColor: getRoadRiskColor(smartRoute.safety.level) }
          ]}>
            <Text style={styles.roadRiskLabel}>ROAD RISK</Text>
            <Text style={styles.roadRiskLevel}>
              {smartRoute.safety.level.toUpperCase()}
            </Text>
            <Text style={styles.roadRiskScore}>
              {smartRoute.safety.score}/100
            </Text>
          </View>

          {/* Weather Summary */}
          <View style={styles.weatherSummary}>
            <Text style={styles.weatherTitle}>Weather Forecast</Text>
            <View style={styles.weatherRow}>
              <View style={styles.weatherPoint}>
                <Text style={styles.weatherEmoji}>
                  {getWeatherEmoji(smartRoute.weather.start.condition)}
                </Text>
                <Text style={styles.weatherTemp}>
                  {smartRoute.weather.start.temperature}°F
                </Text>
                <Text style={styles.weatherLabel}>Start</Text>
              </View>
              <View style={styles.weatherPoint}>
                <Text style={styles.weatherEmoji}>
                  {getWeatherEmoji(smartRoute.weather.middle.condition)}
                </Text>
                <Text style={styles.weatherTemp}>
                  {smartRoute.weather.middle.temperature}°F
                </Text>
                <Text style={styles.weatherLabel}>Middle</Text>
              </View>
              <View style={styles.weatherPoint}>
                <Text style={styles.weatherEmoji}>
                  {getWeatherEmoji(smartRoute.weather.end.condition)}
                </Text>
                <Text style={styles.weatherTemp}>
                  {smartRoute.weather.end.temperature}°F
                </Text>
                <Text style={styles.weatherLabel}>End</Text>
              </View>
            </View>
          </View>

          {/* Warnings */}
          {smartRoute.safety.warnings.length > 0 && (
            <View style={styles.warningsContainer}>
              <Text style={styles.warningsTitle}>⚠️ Warnings</Text>
              {smartRoute.safety.warnings.map((warning, index) => (
                <Text key={index} style={styles.warningText}>
                  • {warning}
                </Text>
              ))}
            </View>
          )}

          {/* Recommendations */}
          {smartRoute.safety.recommendations.length > 0 && (
            <View style={styles.recommendationsContainer}>
              <Text style={styles.recommendationsTitle}>💡 Recommendations</Text>
              {smartRoute.safety.recommendations.slice(0, 2).map((rec, index) => (
                <Text key={index} style={styles.recommendationText}>
                  • {rec}
                </Text>
              ))}
            </View>
          )}

          {/* AI Badge */}
          <View style={styles.aiBadge}>
            <Text style={styles.aiBadgeText}>🤖 AI Optimized</Text>
          </View>

          {/* Start Navigation Button */}
          <TouchableOpacity style={styles.startButton}>
            <Text style={styles.startButtonText}>Start Navigation</Text>
          </TouchableOpacity>
        </View>
      )}

      {/* Refresh Button */}
      <TouchableOpacity
        style={styles.refreshButton}
        onPress={loadSmartRoute}
      >
        <Text style={styles.refreshButtonText}>🔄 Refresh Route</Text>
      </TouchableOpacity>
    </View>
  );
}
